# 第三方包的使用

### Cropperjs 图片裁剪

1、安装cropperjs

```js
npm install cropperjs
```

2、引入css、js

```js
import 'cropperjs/dist/cropper.css'
import Cropper from 'cropperjs'
```

3、在`mounted`中初始化
首先在模板中添加img的引用

```html
<div>
    <img
      class="img"
      :src="img"
      ref="img"
    />
</div>

追加样式

.img {
+  display: block;
   max-width: 100%;
}
```

```javascript
mounted () {
//在monunted初始化cropperjs
    const image = this.$refs.img
    this.cropper = new Cropper(image, {  //crop剪裁，Cropper修剪机
      aspectRatio: 16 / 9,
      crop (event) {
        console.log(event.detail.x)
        console.log(event.detail.y)
        console.log(event.detail.width)
        console.log(event.detail.height)
        console.log(event.detail.rotate)
        console.log(event.detail.scaleX)
        console.log(event.detail.scaleY)
      }
    })
    console.log(cropper)
  }
```

4. 配置cropperjs

```js
const cropper = new Cropper(image, {
    viewMode: 1, // 只能在裁剪的图片范围内移动
    dragMode: 'move', // 画布可以移动
    aspectRatio: 1, // 裁剪区默认正方形
    autoCropArea: 1, // 自动调整裁剪图片
    cropBoxMovable: false, // 禁止裁剪区移动
    cropBoxResizable: false, // 禁止裁剪区缩放
    background: false // 关闭默认背景
})
```

5. 使用cropper裁剪后的图片
  
   ```javascript
   this.cropper.getCroppedCanvas().toBlob((blob) => {
                   //blob就是裁剪后的图片的地址
           //调用需要图片的函数把地址传给他
           this.updatePhtot(blob)
               })
   ```

### Dayjs 时间处理

1、安装

```sh
npm i dayjs
```

2、创建 `utils/dayjs.js`

```js
import Vue from 'vue'
import dayjs from 'dayjs'

// 加载中文语言包
import 'dayjs/locale/zh-cn'
//导入处理相对时间的插件
import relativeTime from 'dayjs/plugin/relativeTime'

// 配置使用处理相对时间的插件
dayjs.extend(relativeTime)

// 配置使用中文语言包
dayjs.locale('zh-cn')

// 全局过滤器：处理相对时间
Vue.filter('relativeTime', value => {
  return dayjs().to(dayjs(value))
})
```

3、在 `main.js` 中加载初始化

```js
import './utils/dayjs'
```

4、使用

使用过滤器：

```html
<p>{{ 日期数据 | relativeTime }}</p>
```

### swiper轮播图（vue2中使用）

1. 安装

   ```shell
   npm install swiper@5
   ```

2. 引入

   ```javascript
   import Swiper from 'swiper'
   import 'swiper/css/swiper.min.css'
   ```

3. 使用（为了防止同一个页面使用多个轮播冲突,创建swiper时使用自定义的的类名，或id,第二个swiper实例创建时使用setTimeout()延时创建，解决冲突问题）

   ```javascript
   <div class="swiper-container banner_swiper">
       <div class="swiper-wrapper">
           <div class="swiper-slide">Slide 1</div>
           <div class="swiper-slide">Slide 2</div>
           <div class="swiper-slide">Slide 3</div>
       </div>
       <!-- 如果需要分页器 -->
       <div class="swiper-pagination"></div>
       
       <!-- 如果需要导航按钮 -->
       <div class="swiper-button-prev"></div>
       <div class="swiper-button-next"></div>
       
       <!-- 如果需要滚动条 -->
       <div class="swiper-scrollbar"></div>
   </div>
   导航等组件可以放在Swiper容器之外
   
   ================================================================
   mounted(){
       //在dom挂载更新后在初始化
       this.$nextTick(this.newswiper())
   },
   methods: {
       newswiper() {
         var mySwiper = new Swiper('.banner_swiper', {
           direction: 'vertical', // 垂直切换选项
           // mousewheel: true, //鼠标滚动切换
           slidesPerView: 'auto', //设置slids根据内容自适应高度
           // loopedSlides: '1', //设置轮播每次显示的数量
           autoHeight: true, //高度随内容变化
           loop: true, // 循环模式选项
           autoplay: { stopOnLastSlide: true },
           // 如果需要分页器
           pagination: {
             el: '.swiper-pagination',
             clickable: true,
             renderBullet: function (index, className) {
               return `<span class='${className} active'></span>`
             }
           }
         })
       }
     }
   ```

4. 轮播图小BUG,不会自动播放，怎么修改配置也没用，想了一个歪路子

```javascript
 let options = {
          spaces: 0,
          ignoreComment: true,
          compact: true,
          declarationKey: 'declaration',
          attributesKey: 'attributes',
          textKey: 'text',
          parentKey: 'parent',
          doctypeKey: 'doctype',

let text = e.target.files[0]
 var reader = new FileReader()
 reader.readAsText(text, 'gb2312')

          //将读取的字符串转换为js对象
          let result = convert.xml2js(e.target.result, options)

          let ceshi = convert.js2xml(result, options)
        }

	```

	```javascript
	import iconv form 'iconv-lite'
	```

3. 使用

	```javascript
	let text = e.target.files[0]
	        var reader = new FileReader()
	        reader.readAsText(text, 'gb2312')
	        reader.onload = function (e) {
	          //将读取的字符串转换为js对象
	          let result = convert.xml2js(e.target.result, options)
	          console.log(result)
	          //将js对象转换为xml格式字符串
	          let ceshi = convert.js2xml(result, options)
	          //将字符串进行编码gb2312
	          let ceshi2 = iconv.encode(ceshi, 'GB2312')
	          console.log(ceshi2)
	          let a = document.createElement('a')
	          //将编码后的typeBuffer取出原始Buffer给blod创建文件类型
	          let blob = new Blob([ceshi2.buffer], {
	            type: 'application/xml',
	            endings: 'native'
	          })
	          a.href = URL.createObjectURL(blob)
	          a.download = '测试数据.prj'
	          a.click()
	        }
	```

### snapdom 和 jsPDF 实现高质量的 HTML 转 PDF 功能

#### 基础用法

```javascript
// 截图
const capture = await snapdom(element, {  scale: 2,      // 2倍清晰度  
                                        quality: 0.95  // PNG 质量
                                       });
// 输出方式
const canvas = await capture.toCanvas();  // Canvas 元素
const imgEl = await capture.toPng();      // <img> 元素，src 为 data URL
const svgStr = await capture.toSvg();     // SVG 字符串
```

jsPDF 是最流行的 JavaScript PDF 生成库，支持在浏览器端直接创建 PDF 文件

```javascript
// 创建 PDF 实例
const pdf = new jsPDF({  orientation: 'portrait',  // 纵向  
                       unit: 'mm',               // 单位：毫米  
                       format: 'a4',             // A4 纸张  
                       compress: true            // 启用压缩
                      });

// 添加图片
pdf.addImage(  imageDataUrl,  // Base64 图片数据  
             'PNG',         // 图片格式  
             10,            // X 坐标（mm）  
             10,            // Y 坐标（mm）  
             190,           // 宽度（mm）  
             100            // 高度（mm）);
             
// 添加新页面
pdf.addPage();             
pdf.save('output.pdf')
```
